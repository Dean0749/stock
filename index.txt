const STORE_KEY = 'lhstock-v1';

function saveDB() {
  try {
    localStorage.setItem(STORE_KEY, JSON.stringify(db));
    document.getElementById('db-status').textContent = '已儲存 ' + new Date().toLocaleTimeString('zh-TW',{hour:'2-digit',minute:'2-digit'});
  } catch(e) { showToast('儲存失敗'); }
}

function loadDB() {
  try {
    const raw = localStorage.getItem(STORE_KEY);
    if (raw) {
      const parsed = JSON.parse(raw);
      db = { ...db, ...parsed };
    }
  } catch(e) {}
}

function exportJSON() {
  const blob = new Blob([JSON.stringify(db, null, 2)], { type:'application/json' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a'); a.href = url; a.download = 'lhstock-db.json'; a.click();
}

function importJSON() {
  const input = document.createElement('input'); input.type = 'file'; input.accept = '.json';
  input.onchange = e => {
    const file = e.target.files[0]; if (!file) return;
    const reader = new FileReader();
    reader.onload = ev => {
      try {
        db = JSON.parse(ev.target.result);
        saveDB(); renderAll(); showToast('匯入成功');
      } catch { showToast('格式錯誤'); }
    };
    reader.readAsText(file);
  };
  input.click();
}

請根據以下維度，從台股中精選 5 支值得關注的股票：

{

}`;

      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1500,
        tools: [{ type: 'web_search_20250305', name: 'web_search' }],
        messages: [{ role: 'user', content: prompt }]
      })
    });
    const data = await res.json();
    const text = (data.content || []).filter(b => b.type==='text').map(b => b.text).join('');
    const s = text.indexOf('{'), e = text.lastIndexOf('}');
    if (s === -1) throw new Error('no json');
    const json = JSON.parse(text.slice(s, e+1));

    loading.style.display = 'none';

    if (json.market_summary) {
      mbar.textContent = '📊 ' + json.market_summary;
      mbar.style.display = 'block';
    }

    (json.picks || []).forEach(s => renderStockCard(container, s));

    // auto-save to log
    const today_iso = new Date().toISOString().slice(0,10);
    const names = (json.picks||[]).map(p => p.name).join('、');
    const existing = db.picksLog.findIndex(p => p.date === today_iso);
    if (existing === -1) {
      db.picksLog.unshift({ date: today_iso, stocks: names, result: null, ts: new Date().toISOString() });
      if (db.picksLog.length > 50) db.picksLog = db.picksLog.slice(0,50);

    btn.innerHTML = '★ 重新分析';
    btn.disabled = false;
  }
}

function addPickLog() {
  const date=val('pk-date'), result=parseFloat(val('pk-result')), stocks=val('pk-stocks');

  db.picksLog.unshift({ date:date||new Date().toISOString().slice(0,10), result:isNaN(result)?null:result, stocks, ts:new Date().toISOString() });
  if (db.picksLog.length>50) db.picksLog=db.picksLog.slice(0,50);
  saveDB(); clr('pk-date','pk-result','pk-stocks'); renderPicksAdmin(); showToast('已儲存');
}

function renderPicksAdmin() {
  const logs = db.picksLog;
  document.getElementById('pk-total').textContent = logs.length;
  const valid = logs.filter(l=>l.result!=null);
  if (valid.length) {
    const wr = Math.round(valid.filter(l=>l.result>0).length/valid.length*100);
    const best = Math.max(...valid.map(l=>l.result));
    document.getElementById('pk-wr').textContent = wr + '%';
    document.getElementById('pk-best').textContent = '+' + best + '%';
  }
  const allS = logs.flatMap(l=>(l.stocks||'').split(/[,、，]/).map(s=>s.trim()).filter(Boolean));
  const freq = {}; allS.forEach(s=>freq[s]=(freq[s]||0)+1);
  const top = Object.entries(freq).sort((a,b)=>b[1]-a[1])[0];
  document.getElementById('pk-repeat').textContent = top ? top[0] : '—';

  const el = document.getElementById('picks-admin-list');
  if (!logs.length) { el.innerHTML = '<div class="empty">尚無紀錄</div>'; return; }
  el.innerHTML = logs.slice(0,50).map((p,i) => `
    <div class="hist-row">
      <span class="hist-date">${p.date}</span>
      <span class="hist-stocks">${p.stocks}</span>
      ${p.result!=null ? `<span class="${p.result>0?'hist-result-up':'hist-result-dn'}">${p.result>0?'+':''}${p.result}%</span>` : '<span style="color:var(--text3);font-size:12px;">未填結果</span>'}
      <button class="btn sm" onclick="removeItem('picksLog',${i})" style="margin-left:auto;">✕</button>
    </div>`).join('');
}
